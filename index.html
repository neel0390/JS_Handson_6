         var Minimal_Angle = (h, m) => 
        {
        let hA=((h*360)/12+(m*360)/(12*60));
        let mA=((m*360)/60);
        
        let angle=Math.abs(hA-mA);
        if(angle>180){
            angle=360-angle;
        }
        return angle;
        };

Perfect Number Check.

         var Perfect_Check = (N) => 
         {
             let sum = 0;
             for(let i=1;i<=N/2;i++){
                 if(N%i===0){
                 sum +=i;
                 }
             }
             if(sum===N){
                 return "YES";
             }else{
                 return "NO";
             }
         };

Reverse a Number.

       var Reverse_Number = (N) => 
        {
        let val1 =N.toString();
        
        let val2=val1.split('');
        val2.reverse();
        let rev='';
        for(i=0; i< val2.length;i++){
        rev += val2[i];
        }
        var num=parseInt(rev);
        return num;
        };

Reverse the String

            var Reverse_String = (str) => 
        {
            const temp = [];
            let reveseStr ='';
            for(let i=str.length-1; i>=0; i--){
            temp.push(str.charAt(i));
            }
            for(i=0; i< temp.length;i++){
        reveseStr += temp[i];
            }
        return reveseStr;
        };
